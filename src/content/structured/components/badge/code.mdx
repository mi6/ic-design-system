export const ToggleBadge = () => {
  const [showBadge, setShowBadge] = useState(false);
  const handleShowHideBadge = () => {
    setShowBadge(!showBadge);
  };
  return (
    <div style={{ padding: "16px" }}>
      <IcButton variant="secondary" onClick={handleShowHideBadge}>
        <IcBadge
          variant="info"
          textLabel="1"
          slot="badge"
          visible={showBadge}
        />
        Hide/Show badge
      </IcButton>
    </div>
  );
};
