  const [data, setData] = useState(LONG_DATA);
  const [itemsPerPage, setItemsPerPage] = useState(5);
  const [dataStart, setDataStart] = useState(0);
  const [updateData, setUpdateData] = useState(false);
  const icItemsPerPageChangeHandler = (ev) => {
    setItemsPerPage(ev.detail.value);
    setDataStart(0);
    setUpdateData(true);
  };
  const pageChangeHandler = (ev) => {
    if (ev.detail.fromItemsPerPage !== true) {
      setDataStart((ev.detail.value - 1) * itemsPerPage);
      setUpdateData(true);
    }
  };
  useEffect(() => {
    if (updateData) {
      setData(LONG_DATA.slice(dataStart, dataStart + itemsPerPage));
      setUpdateData(false);
    }
  }, [updateData, itemsPerPage, dataStart]);
  return (
    <IcDataTable
      caption="Slotted pagination bar"
      columns={LONG_COLUMNS}
      data={data}
    >
      <IcPaginationBar
        slot="pagination-bar"
        showItemsPerPageControl
        totalItems={LONG_DATA.length}

          { label: "5", value: "5" },
          { label: "10", value: "10" },
          { label: "15", value: "15" },
        ]}
        selectedItemsPerPage={itemsPerPage}
        setToFirstPageOnPaginationChange
        onIcItemsPerPageChange={icItemsPerPageChangeHandler}
        onIcPageChange={pageChangeHandler}
      />
    </IcDataTable>
